### django python shell에서 데이터 입력하기, 조회하기
```s
# username과 비밀번호 입력하기 (가입)
Users.objects.create(username='test1', password='test1234')
# 출력값 <Users: Users object (1)>

# 특정 id값(PK)을 가진 객체를 쿼리셋 형태로 불러오기
Users.objects.filter(id=2)
# <QuerySet [<Users: Users object (2)>]>

# 그것을 객체 형태로 발라내기
Users.objects.filter(id=2)[0]
# <Users: Users object (2)>

# 특정 username을 가진 유저 정보를 불러와 쿼리셋 형태로 유저 정보 조회하기
Users.objects.filter(username='test3').values()
# <QuerySet [{'id': 3, 'username': 'test3', 'password': 'test1234'}]>

# 그것을 객체 형태로 발라내기
Users.objects.filter(username='test2').values()[0]
# {'id': 2, 'username': 'test2', 'password': 'test1234'}

# 특정 username을 가진 유저 정보가 담긴 객체를 불러와 username이 무엇인지 조회하기
Users.objects.filter(username='test2').values()[0]['password']
# 'test1234'

# 위 작업을 변수 할당 방식으로 간편하게 할 수도 있다.
user_id = Users.objects.get(username='test1')
user_id.password
# 'test1234'

# users 테이블과 Foreign key로 연결한 Comments 테이블에 데이터 입력하기(댓글 입력)
Comments.objects.create(username=Users.objects.filter(id=2)[0], comment="he he")
# <Comments: Comments object (1)>
# 댓글 등록됨
```
